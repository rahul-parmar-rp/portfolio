## JavaScript Practice

### Sum nested salary values

```javascript
const company = {
  sales: [
    { name: "John", salary: 1000 },
    { name: "Alice", salary: 1600 },
  ],
  development: {
    sites: [
      { name: "Peter", salary: 2000 },
      { name: "Alex", salary: 1800 },
    ],
    internals: [{ name: "Jack", salary: 1300 }],
  },
};

function findSum(obj) {
  let sumOfSalary = 0;
  Object.keys(obj).forEach((k) => {
    const child = obj[k];
    if (Array.isArray(child) && child.length > 0) {
      child.forEach((person) => {
        sumOfSalary = sumOfSalary + person.salary;
      });
    }
    if (child && typeof child === "object" && !Array.isArray(child)) {
      sumOfSalary = sumOfSalary + findSum(child);
    }
  });
  return sumOfSalary;
}

console.log(findSum(company));
```

### Count number occurrences

```javascript
function findOccurences(nums) {
  const result = {};
  for (let index = 0; index < nums.length; index++) {
    const element = nums[index];
    if (result[element]) {
      result[element] = result[element] + 1;
    } else {
      result[element] = 1;
    }
  }
  return result;
}

console.log(findOccurences([10, 20, 5, 10, 1, 2, 11, 1, 2, 5, 5, 55]));
```

### Find smallest number in array

```javascript
const numbersArr = [10, -1, 5, -5, 0];

function findMin(nums) {
  let min = Infinity;
  for (let index = 0; index < nums.length; index++) {
    if (min > nums[index]) {
      min = nums[index];
    }
  }
  return min;
}

console.log(findMin(numbersArr));
```

### Reverse words in a sentence

```javascript
const example = "This is your frontend design to implement with developer";

function reverseWords(testStr) {
  const words = [];
  let word = "";

  for (let index = 0; index < testStr.length; index++) {
    const char = testStr[index];
    if (char !== " " || index === testStr.length - 1) {
      word = word.concat(char);
    } else {
      words.push(word);
      word = "";
    }

    if (index === testStr.length - 1) {
      words.push(word);
      word = "";
    }
  }

  let strg = "";
  for (let index = words.length - 1; index >= 0; index--) {
    strg = strg.concat(words[index]) + (index !== 0 ? " " : "");
  }

  return strg;
}

console.log(reverseWords(example));
```
