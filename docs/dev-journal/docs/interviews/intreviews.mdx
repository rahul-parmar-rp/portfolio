import { useState, useEffect } from "react";
import { UseCountryDataProps, Country } from "../types";
import { fetchCountryData } from "../services/countryService";


export const useCountryData = ({ query }: UseCountryDataProps) => {
  const [data, setData] = useState<Country[]>([]);
  const [loading, setLoading] = useState<boolean>(false);
  const [error, setError] = useState<string | null>(null);
  const [hasSearched, setHasSearched] = useState<boolean>(false);


  useEffect(() => {
    const fetchData = async () => {
      if (!query) {
        setData([]);
        setHasSearched(false);
        return;
      }


      setLoading(true);
      setHasSearched(true);
      setError(null);
      try {
        const fetchedData = await fetchCountryData(query);
        setData(fetchedData);
      } catch (error) {
        setError("Error fetching countries");
      } finally {
        setLoading(false);
      }
    };


    fetchData();
  }, [query]);


  return { data, loading, error, hasSearched };
};

function findLeaders(arr) {
    const leaders = [];
    let maxRight = arr[arr.length - 1]; // Start with the last element as the maximum

    // Iterate the array from right to left
    for (let i = arr.length - 2; i >= 0; i--) {
        if (arr[i] > maxRight) {
            leaders.unshift(arr[i]); // Add the leader to the beginning of the leaders array
